2.  player 선수중 키가 170 미만이거나 194 초과인 
    선수들만   이름, 키 조회하기
1) select player_name, height
   from player
   where height < 170 OR  height > 194; 
2) select player_name, height
   from player
   where height NOT between 170 and 194 ;

1) select  player_name, e_player_name
   from player

select player_name, e_player_name from player
where e_player_name like '%e%e%'  
  AND e_player_name NOT like '%e%e%e%'  ; 

4. GK포지션인 선수중 
   소속팀이 K01팀이거나 키가 185보다 작은 선수들만
   이름, 포지션,팀id,키 조회하기.
1> 2 * 3 +4 
select player_name, position, team_id
from player
where POSITION='GK' AND team_id='K01' OR height < 185; 
2> 2 * (3+4)
select player_name, position, team_id
from player
where POSITION='GK' AND (team_id='K01' OR height < 185);
  
5. player선수중 포지션이 GK,MF,FW,DF 모두 아닌 선수들만
   이름, 포지션조회하기. 

1) select player_name, position 
   from player
    where position  <> 'GK' AND
          position  <> 'MF' AND
          position  <> 'FW' AND
          position <> 'DF' ; 
2) 
 select player_name, position 
   from player
